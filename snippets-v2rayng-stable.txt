function parseAddressAndPort(input) {
  if (!input) return { address: '', port: null };
  if (input.includes('[') && input.includes(']')) {
    const match = input.match(/^\[([^\]]+)\](?::(\d+))?$/);
    if (match) {
      return { address: match[1], port: match[2] ? parseInt(match[2], 10) : null };
    }
  }
  const lastColonIndex = input.lastIndexOf(':');
  if (lastColonIndex > 0) {
    const address = input.substring(0, lastColonIndex);
    const portStr = input.substring(lastColonIndex + 1);
    const port = parseInt(portStr, 10);
    if (!isNaN(port) && port > 0 && port <= 65535) return { address, port };
  }
  return { address: input, port: null };
}

function parseUrlCustomIPs(input) {
  if (!input) return [];
  const out = [];
  const seen = new Set();
  const lines = String(input).replace(/\r/g, '').split(/[\n,]+/);
  for (let i = 0; i < lines.length; i++) {
    let line = lines[i].trim();
    if (!line) continue;
    let addr = line;
    let name = '自定义IP-' + (out.length + 1);
    const h = line.indexOf('#');
    if (h >= 0) {
      addr = line.slice(0, h).trim();
      name = line.slice(h + 1).trim() || name;
    }
    const p = parseAddressAndPort(addr);
    if (!p.address) continue;
    const port = p.port || 443;
    const key = p.address + ':' + port;
    if (seen.has(key)) continue;
    seen.add(key);
    out.push({ ip: p.address, port, name });
  }
  return out;
}

async function handleSubscriptionRequest(request, uuid, url = null) {

  const customIPsRaw = url.searchParams.get('ips') || '';
  if (customIPsRaw) {
    const customIPList = parseUrlCustomIPs(customIPsRaw);
    if (customIPList.length > 0) finalLinks.push(...generateLinksFromNewIPs(customIPList, uuid, workerDomain));
  }

  if (enableOtherPreferred) {
    const domainList = directDomains.map(d => ({ ip: d.domain, isp: d.name || d.domain }));

  if (enableGitHubPreferred) {
    const newIPList = await fetchAndParseNewIPs();
    if (newIPList.length > 0) finalLinks.push(...generateLinksFromNewIPs(newIPList, uuid, workerDomain));
  }

    const errorRemark = "所有节点获取失败";

  }

  const subscriptionContent = btoa(finalLinks.join('\n'));

  return new Response(subscriptionContent, {
    headers: {
      'Content-Type': 'text/plain; charset=utf-8',
      'Cache-Control': 'no-store, no-cache, must-revalidate, max-age=0',
    },
  });
}

function generateLinksFromSource(list, uuid, workerDomain) {

      pushWsLinks(links, proto, uuid, safeIP, port, workerDomain, wsPath, wsNodeName);

  return links;
}

async function fetchAndParseNewIPs() {
  const url = githubPreferredURL;
  try {
    const response = await fetch(url);
    if (!response.ok) return [];
    const text = await response.text();
    const results = [];
    const lines = text.trim().replace(/\r/g, "").split('\n');
    const regex = /^([^:]+):(\d+)#(.*)$/;

    for (const line of lines) {
      const trimmedLine = line.trim();
      if (!trimmedLine) continue;
      const match = trimmedLine.match(regex);
      if (match) {
        results.push({
          ip: match[1],
          port: parseInt(match[2], 10),
          name: match[3].trim() || match[1]
        });
        if (maxGitHubPreferredNodes > 0 && results.length >= maxGitHubPreferredNodes) break;
      }
    }
    return results;
  } catch (error) {
    return [];
  }
}

function generateLinksFromNewIPs(list, uuid, workerDomain) {

  const proto = 'vless';

  list.forEach(item => {
    const nodeName = item.name;

}

function pushWsLinks(links, proto, uuid, safeIP, port, workerDomain, wsPath, nodeName) {
  links.push(buildWsLink(proto, uuid, safeIP, port, workerDomain, wsPath, nodeName));
}

function buildWsLink(proto, uuid, safeIP, port, workerDomain, wsPath, nodeName) {
  const wsParams = new URLSearchParams({
    encryption: 'none',
    security: 'tls',
    sni: workerDomain,
    fp: tlsFingerprint,
    type: 'ws',
    host: workerDomain,
    path: wsPath
  });
  return `${proto}://${uuid}@${safeIP}:${port}?${wsParams.toString()}#${encodeURIComponent(nodeName)}`;
}

async function handleWsRequest(request, currentFallbackAddress = null) {
  const wsPair = new WebSocketPair();
  const [clientSock, serverSock] = Object.values(wsPair);
  serverSock.accept();
  serverSock.binaryType = 'arraybuffer';

  let remoteConnWrapper = { socket: null, writer: null, drain: null };
  let isDnsQuery = false, uploadBusy = false, transportClosed = false;
  const uploadQueue = createChunkQueue(UP_PACK, UP_MAX);

  const fbAddr = currentFallbackAddress || fallbackAddress;

  function releaseRemoteWriter() {
    try { remoteConnWrapper.writer?.releaseLock(); } catch (error) { }
    remoteConnWrapper.writer = null;
  }

  function closeTransport() {
    if (transportClosed) return;
    transportClosed = true;
    uploadQueue.clear();
    releaseRemoteWriter();
    closeSocketQuietly(remoteConnWrapper.socket);
    closeSocketQuietly(serverSock);
  }

  function queueUpload(chunk) {
    if (!uploadQueue.push(chunk)) {
      closeTransport();
      return false;
    }
    remoteConnWrapper.drain?.();
    return true;
  }

  async function drainUpload() {
    if (uploadBusy || transportClosed || !remoteConnWrapper.writer) return;
    uploadBusy = true;
    try {
      for (;;) {
        if (transportClosed || !remoteConnWrapper.writer) break;
        const data = uploadQueue.take();
        if (!data) break;
        await remoteConnWrapper.writer.write(data);
      }
    } catch (error) {
      closeTransport();
    } finally {
      uploadBusy = false;
      if (!uploadQueue.empty && !transportClosed && remoteConnWrapper.writer) queueMicrotask(drainUpload);
    }
  }

  remoteConnWrapper.drain = () => {
    if (!uploadBusy && !uploadQueue.empty && remoteConnWrapper.writer) queueMicrotask(drainUpload);
  };

  const earlyData = request.headers.get('sec-websocket-protocol') || '';
  const readable = makeReadableStream(serverSock, earlyData);

  readable.pipeTo(new WritableStream({
    async write(chunk) {
      if (transportClosed) return;
      if (isDnsQuery) return await forwardUDP(chunk, serverSock, null);
      if (remoteConnWrapper.socket && remoteConnWrapper.writer) {
        if (!queueUpload(chunk)) throw new Error('upload queue overflow');
        return;
      }
      const { hasError, message, addressType, port, hostname, rawIndex, version, isUDP } = parseWsPacketHeader(chunk, authToken);
      if (hasError) throw new Error(message);

      if (isUDP) {
        if (port === 53) isDnsQuery = true;
        else throw new Error(E_UDP_DNS_ONLY);
      }
      const respHeader = new Uint8Array([version[0], 0]);
      const rawData = chunk.slice(rawIndex);

      if (isDnsQuery) return forwardUDP(rawData, serverSock, respHeader);
      await forwardTCP(addressType, hostname, port, rawData, serverSock, respHeader, remoteConnWrapper, fbAddr);
    },
  })).catch((err) => { console.log('WS Stream Error:', err); closeTransport(); });

  return new Response(null, { status: 101, webSocket: clientSock });
}

async function forwardTCP(addrType, host, portNum, rawData, ws, respHeader, remoteConnWrapper, fbAddr = null) {
  let retried = false;

  async function getFallbackTarget() {
    if (fbAddr && fbAddr.trim()) {
      const parsed = parseAddressAndPort(fbAddr);
      return { host: parsed.address, port: parsed.port || portNum };
    }
    if (fallbackAddress && fallbackAddress.trim()) {
      const parsed = parseAddressAndPort(fallbackAddress);
      return { host: parsed.address, port: parsed.port || portNum };
    }
    const bestBackupIP = await getBestBackupIP(currentWorkerRegion);
    return bestBackupIP ? { host: bestBackupIP.domain, port: bestBackupIP.port } : { host, port: portNum };
  }

  async function connectAndSend(address, port, fallbackTarget = null) {
    const primary = { host: address, port };
    const shouldRace = fallbackTarget && (fallbackTarget.host !== address || fallbackTarget.port !== port);
    const remoteSock = shouldRace ? await raceTcpSocket(primary, fallbackTarget) : await connectTcpSocket(address, port);
    const writer = remoteSock.writable.getWriter();
    try {
      if (rawData.byteLength) await writer.write(toUint8Array(rawData));
      return { remoteSock, writer };
    } catch (error) {
      try { writer.releaseLock(); } catch (e) { }
      closeSocketQuietly(remoteSock);
      throw error;
    }
  }

  function attachRemote(remoteSock, writer, retryFunc) {
    try {
      if (remoteConnWrapper.writer && remoteConnWrapper.writer !== writer) remoteConnWrapper.writer.releaseLock();
    } catch (error) { }
    remoteConnWrapper.socket = remoteSock;
    remoteConnWrapper.writer = writer;
    remoteConnWrapper.drain?.();
    remoteSock.closed.catch(() => { }).finally(() => {
      if (remoteConnWrapper.socket === remoteSock && (!retryFunc || !retried)) closeSocketQuietly(ws);
    });
    void connectStreams(remoteSock, ws, respHeader, retryFunc).finally(() => {
      if (remoteConnWrapper.socket === remoteSock) {
        try { writer.releaseLock(); } catch (error) { }
        remoteConnWrapper.writer = null;
      }
    });
  }

  async function retryConnection() {
    if (retried || ws.readyState !== 1) return;
    retried = true;
    const fallbackTarget = await getFallbackTarget();
    try {
      const { remoteSock, writer } = await connectAndSend(fallbackTarget.host || host, fallbackTarget.port);
      attachRemote(remoteSock, writer, null);
    } catch (err) {
      console.log('Fallback connection failed:', err);
      closeSocketQuietly(ws);
    }
  }

  try {
    const fallbackTarget = await getFallbackTarget();
    const { remoteSock, writer } = await connectAndSend(host, portNum, fallbackTarget);
    attachRemote(remoteSock, writer, retryConnection);
  } catch (err) {
    console.log('Initial connection failed, trying fallback:', err);
    await retryConnection();
  }
}

function parseWsPacketHeader(chunk, token) {
  if (chunk.byteLength < 24) return { hasError: true, message: E_INVALID_DATA };
  const version = new Uint8Array(chunk.slice(0, 1));
  if (formatIdentifier(new Uint8Array(chunk.slice(1, 17))) !== token) return { hasError: true, message: E_INVALID_USER };
  const optLen = new Uint8Array(chunk.slice(17, 18))[0];
  const cmd = new Uint8Array(chunk.slice(18 + optLen, 19 + optLen))[0];
  let isUDP = false;
  if (cmd === 1) { }
  else if (cmd === 2) { isUDP = true; }
  else { return { hasError: true, message: E_UNSUPPORTED_CMD }; }

  const portIdx = 19 + optLen;
  const port = new DataView(chunk.slice(portIdx, portIdx + 2)).getUint16(0);
  let addrIdx = portIdx + 2, addrLen = 0, addrValIdx = addrIdx + 1, hostname = '';
  const addressType = new Uint8Array(chunk.slice(addrIdx, addrValIdx))[0];

  switch (addressType) {
    case ADDRESS_TYPE_IPV4:
      addrLen = 4;
      hostname = new Uint8Array(chunk.slice(addrValIdx, addrValIdx + addrLen)).join('.');
      break;
    case ADDRESS_TYPE_URL:
      addrLen = new Uint8Array(chunk.slice(addrValIdx, addrValIdx + 1))[0];
      addrValIdx += 1;
      hostname = new TextDecoder().decode(chunk.slice(addrValIdx, addrValIdx + addrLen));
      break;
    case ADDRESS_TYPE_IPV6:
      addrLen = 16;
      const ipv6 = [];
      const ipv6View = new DataView(chunk.slice(addrValIdx, addrValIdx + addrLen));
      for (let i = 0; i < 8; i++) ipv6.push(ipv6View.getUint16(i * 2).toString(16));
      hostname = ipv6.join(':');
      break;
    default:
      return { hasError: true, message: `${E_INVALID_ADDR_TYPE}: ${addressType}` };
  }

  if (!hostname) return { hasError: true, message: `${E_EMPTY_ADDR}: ${addressType}` };
  return { hasError: false, addressType, port, hostname, isUDP, rawIndex: addrValIdx + addrLen, version };
}

function makeReadableStream(socket, earlyDataHeader) {
  let cancelled = false;
  return new ReadableStream({
    start(controller) {
      socket.addEventListener('message', (event) => { if (!cancelled) controller.enqueue(event.data); });
      socket.addEventListener('close', () => { if (!cancelled) { closeSocketQuietly(socket); controller.close(); } });
      socket.addEventListener('error', (err) => controller.error(err));
      const { earlyData, error } = base64ToArray(earlyDataHeader);
      if (error) controller.error(error);
      else if (earlyData) controller.enqueue(earlyData);
    },
    cancel() { cancelled = true; closeSocketQuietly(socket); }
  });
}

async function connectStreams(remoteSocket, webSocket, headerData, retryFunc) {
  let header = headerData, hasData = false, retried = false, reader = null, byob = true;
  let timer = retryFunc ? setTimeout(() => {
    if (!hasData && !retried) {
      retried = true;
      closeSocketQuietly(remoteSocket);
      void retryFunc();
    }
  }, FIRST_BYTE_TIMEOUT) : null;
  try {
    try {
      reader = remoteSocket.readable.getReader({ mode: 'byob' });
    } catch (error) {
      byob = false;
      reader = remoteSocket.readable.getReader();
    }
    for (;;) {
      const result = byob ? await reader.read(new Uint8Array(T_CHUNK)) : await reader.read();
      if (result.done) break;
      const chunk = toUint8Array(result.value);
      if (!chunk.byteLength) continue;
      if (!hasData) {
        hasData = true;
        if (timer) { clearTimeout(timer); timer = null; }
      }
      if (webSocket.readyState !== 1) throw new Error(E_WS_NOT_OPEN);
      if (header) {
        webSocket.send(await new Blob([header, chunk]).arrayBuffer());
        header = null;
      } else {
        webSocket.send(chunk);
      }
    }
  } catch (error) {
    console.error("Stream connection error:", error);
    if (!hasData && retryFunc && !retried && webSocket.readyState === 1) {
      retried = true;
      await retryFunc();
      return;
    }
    if (hasData || !retryFunc || webSocket.readyState !== 1) closeSocketQuietly(webSocket);
  } finally {
    if (timer) clearTimeout(timer);
    try { reader?.releaseLock(); } catch (error) { }
  }

  if (!hasData && retryFunc && !retried && webSocket.readyState === 1) await retryFunc();
}

async function forwardUDP(udpChunk, webSocket, respHeader) {
  try {
    const tcpSocket = await connectTcpSocket('8.8.4.4', 53);
    let vlessHeader = respHeader;
    await writeToSocket(tcpSocket, udpChunk);
    await tcpSocket.readable.pipeTo(new WritableStream({
      async write(chunk) {
        if (webSocket.readyState === 1) {
          if (vlessHeader) {
            webSocket.send(await new Blob([vlessHeader, chunk]).arrayBuffer());
            vlessHeader = null;
          } else {
            webSocket.send(chunk);
          }
        }
      },
    }));
  } catch (error) {
    console.error(`DNS forward error: ${error.message}`);
  }
}

function base64ToArray(b64Str) {
  if (!b64Str) return { error: null };
  try {
    b64Str = b64Str.replace(/-/g, '+').replace(/_/g, '/');
    return { earlyData: Uint8Array.from(atob(b64Str), (c) => c.charCodeAt(0)).buffer, error: null };
  } catch (error) {
    return { error };
  }
}

async function writeToSocket(socket, chunk) {
  if (!socket || !chunk || chunk.byteLength === 0) return;
  const writer = socket.writable.getWriter();
  try {
    await writer.write(toUint8Array(chunk));
  } finally {
    try { writer.releaseLock(); } catch (error) { }
  }
}

async function connectTcpSocket(hostname, port) {
  const socket = connect({ hostname, port });
  try {
    await withTimeout(socket.opened, TCP_TIMEOUT, 'TCP connect timeout');
    return socket;
  } catch (error) {
    closeSocketQuietly(socket);
    throw error;
  }
}

function raceTcpSocket(primary, backup) {
  let done = false, fail = 0, timer;
  return new Promise((resolve, reject) => {
    const start = (target) => connectTcpSocket(target.host, target.port).then(socket => {
      if (done) return closeSocketQuietly(socket);
      done = true;
      clearTimeout(timer);
      resolve(socket);
    }, error => {
      if (++fail >= 2 && !done) reject(error);
    });
    start(primary);
    timer = setTimeout(() => start(backup), RACE_DELAY);
  });
}

function withTimeout(promise, ms, message) {
  let timer;
  return Promise.race([
    promise.finally(() => clearTimeout(timer)),
    new Promise((_, reject) => { timer = setTimeout(() => reject(new Error(message)), ms); })
  ]);
}

function createChunkQueue(pack, max) {
  let queue = [], head = 0, size = 0;
  return {
    get empty() { return size === 0; },
    clear() { queue = []; head = 0; size = 0; },
    push(chunk) {
      const data = toUint8Array(chunk);
      if (!data.byteLength) return true;
      if (size + data.byteLength > max) return false;
      for (let i = 0; i < data.byteLength; i += pack) queue.push(data.subarray(i, i + pack));
      size += data.byteLength;
      return true;
    },
    take() {
      if (head >= queue.length) return null;
      const chunks = [];
      let len = 0;
      while (head < queue.length && len < pack) {
        const data = queue[head++];
        chunks.push(data);
        len += data.byteLength;
        size -= data.byteLength;
      }
      if (head > 64 && head * 2 > queue.length) {
        queue = queue.slice(head);
        head = 0;
      }
      return joinChunks(chunks, len);
    }
  };
}

function joinChunks(chunks, len) {
  if (chunks.length === 1) return chunks[0];
  const out = new Uint8Array(len);
  let offset = 0;
  for (const chunk of chunks) {
    out.set(chunk, offset);
    offset += chunk.byteLength;
  }
  return out;
}

function toUint8Array(value) {
  if (value instanceof Uint8Array) return value;
  if (value instanceof ArrayBuffer) return new Uint8Array(value);
  return new Uint8Array(value.buffer, value.byteOffset, value.byteLength);
}

function isValidFormat(uuid) {
  return /^[0-9a-f]{8}-[0-9a-f]{4}-4[0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i.test(uuid);
}

function closeSocketQuietly(socket) {
  try {
    if (!socket || typeof socket.close !== 'function') return;
    if (socket.readyState === undefined || socket.readyState === 1 || socket.readyState === 2) socket.close();
  }
  catch (error) { }
}

const hexTable = Array.from({ length: 256 }, (v, i) => (i + 256).toString(16).slice(1));

function formatIdentifier(arr, offset = 0) {
  const id = (
    hexTable[arr[offset]] + hexTable[arr[offset + 1]] + hexTable[arr[offset + 2]] + hexTable[arr[offset + 3]] + "-" +
    hexTable[arr[offset + 4]] + hexTable[arr[offset + 5]] + "-" +
    hexTable[arr[offset + 6]] + hexTable[arr[offset + 7]] + "-" +
    hexTable[arr[offset + 8]] + hexTable[arr[offset + 9]] + "-" +
    hexTable[arr[offset + 10]] + hexTable[arr[offset + 11]] + hexTable[arr[offset + 12]] + hexTable[arr[offset + 13]] +
    hexTable[arr[offset + 14]] + hexTable[arr[offset + 15]]
  ).toLowerCase();
  if (!isValidFormat(id)) throw new TypeError(E_INVALID_ID_STR);
  return id;
}
