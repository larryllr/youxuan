  async fetch(request) {
    try {
      const url = new URL(request.url);
      currentWorkerRegion = manualWorkerRegion ? manualWorkerRegion.trim().toUpperCase() : detectWorkerRegion(request);
      const fallback = getBestBackupAddress(currentWorkerRegion);

      if (request.headers.get('Upgrade') === 'websocket') {

      }

      if (url.pathname === '/') return html('服务正常', '请在网址后输入 UUID 进入订阅页面。');

      const route = getRoute(url.pathname);
      if (!route.ok) return new Response('Not Found', { status: 404 });
      if (route.uuid !== authToken) return new Response('UUID错误', { status: 403 });

      if (route.type === 'page') return subPage(request);
      if (route.type === 'sub') return subContent(request, route.uuid, url);
      return new Response('Not Found', { status: 404 });
    } catch (e) {
      return new Response(String(e), { status: 500 });
    }
  }
};

function getRoute(pathname) {
  const clean = pathname.replace(/\/+$/, '');
  if (customPath && customPath.trim()) {
    const p = '/' + customPath.trim().replace(/^\/+|\/+$/g, '');
    if (clean === p) return { ok: true, type: 'page', uuid: authToken };
    if (clean === p + '/sub') return { ok: true, type: 'sub', uuid: authToken };
    return { ok: false };
  }
  const parts = clean.split('/').filter(Boolean);
  if (parts.length === 1 && isValidFormat(parts[0])) return { ok: true, type: 'page', uuid: parts[0] };
  if (parts.length === 2 && isValidFormat(parts[0]) && parts[1] === 'sub') return { ok: true, type: 'sub', uuid: parts[0] };
  return { ok: false };
}

async function subContent(request, uuid, url) {
  const workerDomain = url.hostname;
  const links = [];

  links.push(...genSource([{ ip: workerDomain, name: '原生地址' }], uuid, workerDomain));

  if (enableOtherPreferred) {
    links.push(...genSource(directDomains.map(x => ({ ip: x, name: x })), uuid, workerDomain));
  }

  });

function parsePreferred(input) {
  return String(input || '').replace(/\r/g, '').split(/[\n,]+/).map((line, i) => {
    line = line.trim();
    if (!line) return null;
    const idx = line.indexOf('#');
    const addr = idx >= 0 ? line.slice(0, idx).trim() : line;
    const name = idx >= 0 ? line.slice(idx + 1).trim() : `自定义IP-${i + 1}`;
    const p = parseAddressAndPort(addr);
    return p.address ? { ip: p.address, port: p.port || 443, name: name || p.address } : null;
  }).filter(Boolean);
}

function uniquePreferred(list) {
  const seen = new Set();
  return list.filter(x => {
    const k = x.ip + ':' + x.port;
    if (seen.has(k)) return false;
    seen.add(k);
    return true;
  });
}

function parseAddressAndPort(input) {
  const v = String(input || '').trim();
  if (!v) return { address: '', port: null };
  const m6 = v.match(/^\[([^\]]+)\](?::(\d+))?$/);
  if (m6) return { address: m6[1], port: m6[2] ? +m6[2] : null };
  if ((v.match(/:/g) || []).length > 1) return { address: v, port: null };
  const i = v.lastIndexOf(':');
  if (i > 0) {
    const p = +v.slice(i + 1);
    if (p > 0 && p <= 65535) return { address: v.slice(0, i), port: p };
  }
  return { address: v, port: null };
}

function genSource(list, uuid, host) {
  return list.map(x => genLink(uuid, x.ip, 443, host, `${String(x.name).replace(/\s/g, '_')}-443-WS-TLS`));
}

function genPreferred(list, uuid, host) {
  return list.map(x => genLink(uuid, x.ip, x.port || 443, host, x.name || x.ip));
}

function genLink(uuid, ip, port, host, name) {
  const safe = ip.includes(':') ? `[${ip}]` : ip;
  const q = new URLSearchParams({ encryption: 'none', security: 'tls', sni: host, fp: 'randomized', type: 'ws', host, path: '/?ed=2048' });
  return `vless://${uuid}@${safe}:${port}?${q.toString()}#${encodeURIComponent(name)}`;
}

function getBestBackupAddress(region) {
  const near = { US:['SG','JP','HK','KR'],SG:['JP','HK','KR','US'],JP:['SG','HK','KR','US'],HK:['SG','JP','KR','US'],KR:['JP','HK','SG','US'],DE:['NL','GB','SE','FI'],SE:['DE','NL','FI','GB'],NL:['DE','GB','SE','FI'],FI:['SE','DE','NL','GB'],GB:['DE','NL','SE','FI'] };
  const all = ['US','SG','JP','HK','KR','DE','SE','NL','FI','GB'];
  const n = near[region] || [];
  const order = [region, ...n, ...all.filter(x => x !== region && !n.includes(x))].filter(Boolean);
  const sorted = order.flatMap(r => backupIPs.filter(x => x.regionCode === r));
  const best = sorted[0] || backupIPs[0];
  return best ? best.domain + ':' + best.port : '';
}

async function handleWsRequest(request, fallback) {
  const wsPair = new WebSocketPair();
  const [clientSock, serverSock] = Object.values(wsPair);
  serverSock.accept();

  let remote = { socket: null };
  let isDns = false;
  const earlyData = request.headers.get('sec-websocket-protocol') || '';
  const readable = makeReadableStream(serverSock, earlyData);

  readable.pipeTo(new WritableStream({
    async write(chunk) {
      if (isDns) return forwardUDP(chunk, serverSock, null);
      if (remote.socket) {
        const w = remote.socket.writable.getWriter();
        await w.write(chunk);
        w.releaseLock();
        return;
      }
      const p = parseWsPacketHeader(chunk, authToken);
      if (p.hasError) throw new Error(p.message);

      if (p.isUDP) {
        if (p.port === 53) isDns = true;
        else throw new Error(E_UDP_DNS_ONLY);
      }

      const respHeader = new Uint8Array([p.version[0], 0]);
      const rawData = chunk.slice(p.rawIndex);
      if (isDns) return forwardUDP(rawData, serverSock, respHeader);
      await forwardTCP(p.hostname, p.port, rawData, serverSock, respHeader, remote, fallback);
    }
  })).catch(e => console.log('WS Stream Error:', e));

  return new Response(null, { status: 101, webSocket: clientSock });
}

async function forwardTCP(host, port, rawData, ws, respHeader, remote, fallback) {
  async function connectAndSend(address, p) {
    const sock = connect({ hostname: address, port: p });
    const w = sock.writable.getWriter();
    await w.write(rawData);
    w.releaseLock();
    return sock;
  }

  async function retry() {
    const fb = parseAddressAndPort(fallback || '');
    const sock = await connectAndSend(fb.address || host, fb.port || port);
    remote.socket = sock;
    sock.closed.catch(() => {}).finally(() => closeSocketQuietly(ws));
    connectStreams(sock, ws, respHeader, null);
  }

  try {
    const sock = await connectAndSend(host, port);
    remote.socket = sock;
    connectStreams(sock, ws, respHeader, retry);
  } catch {
    retry();
  }
}

function parseWsPacketHeader(chunk, token) {
  if (chunk.byteLength < 24) return { hasError: true, message: E_INVALID_DATA };

  const version = new Uint8Array(chunk.slice(0, 1));
  if (formatIdentifier(new Uint8Array(chunk.slice(1, 17))) !== token) {
    return { hasError: true, message: E_INVALID_USER };
  }

  const optLen = new Uint8Array(chunk.slice(17, 18))[0];
  const cmd = new Uint8Array(chunk.slice(18 + optLen, 19 + optLen))[0];
  const isUDP = cmd === 2;
  if (cmd !== 1 && cmd !== 2) return { hasError: true, message: E_UNSUPPORTED_CMD };

  const portIdx = 19 + optLen;
  const port = new DataView(chunk.slice(portIdx, portIdx + 2)).getUint16(0);
  let addrIdx = portIdx + 2;
  let addrValIdx = addrIdx + 1;
  let addrLen = 0;
  let hostname = '';
  const addressType = new Uint8Array(chunk.slice(addrIdx, addrValIdx))[0];

  if (addressType === ADDRESS_TYPE_IPV4) {
    addrLen = 4;
    hostname = new Uint8Array(chunk.slice(addrValIdx, addrValIdx + addrLen)).join('.');
  } else if (addressType === ADDRESS_TYPE_URL) {
    addrLen = new Uint8Array(chunk.slice(addrValIdx, addrValIdx + 1))[0];
    addrValIdx += 1;
    hostname = new TextDecoder().decode(chunk.slice(addrValIdx, addrValIdx + addrLen));
  } else if (addressType === ADDRESS_TYPE_IPV6) {
    addrLen = 16;
    const view = new DataView(chunk.slice(addrValIdx, addrValIdx + addrLen));
    hostname = Array.from({ length: 8 }, (_, i) => view.getUint16(i * 2).toString(16)).join(':');
  } else {
    return { hasError: true, message: `${E_INVALID_ADDR_TYPE}: ${addressType}` };
  }

  if (!hostname) return { hasError: true, message: `${E_EMPTY_ADDR}: ${addressType}` };
  return { hasError: false, port, hostname, isUDP, rawIndex: addrValIdx + addrLen, version };
}

function makeReadableStream(socket, earlyDataHeader) {
  let cancelled = false;
  return new ReadableStream({
    start(controller) {
      socket.addEventListener('message', e => { if (!cancelled) controller.enqueue(e.data); });
      socket.addEventListener('close', () => { if (!cancelled) { closeSocketQuietly(socket); controller.close(); } });
      socket.addEventListener('error', e => controller.error(e));

      const { earlyData, error } = base64ToArray(earlyDataHeader);
      if (error) controller.error(error);
      else if (earlyData) controller.enqueue(earlyData);
    },
    cancel() {
      cancelled = true;
      closeSocketQuietly(socket);
    }
  });
}

async function connectStreams(remoteSocket, webSocket, headerData, retryFunc) {
  let header = headerData;
  let hasData = false;

  await remoteSocket.readable.pipeTo(new WritableStream({
    async write(chunk, controller) {
      hasData = true;
      if (webSocket.readyState !== 1) controller.error(E_WS_NOT_OPEN);

      if (header) {
        webSocket.send(await new Blob([header, chunk]).arrayBuffer());
        header = null;
      } else {
        webSocket.send(chunk);
      }
    }
  })).catch(() => closeSocketQuietly(webSocket));

  if (!hasData && retryFunc) retryFunc();
}

async function forwardUDP(udpChunk, webSocket, respHeader) {
  try {
    const tcpSocket = connect({ hostname: '8.8.4.4', port: 53 });
    let header = respHeader;
    const w = tcpSocket.writable.getWriter();
    await w.write(udpChunk);
    w.releaseLock();

    await tcpSocket.readable.pipeTo(new WritableStream({
      async write(chunk) {
        if (webSocket.readyState !== 1) return;
        if (header) {
          webSocket.send(await new Blob([header, chunk]).arrayBuffer());
          header = null;
        } else {
          webSocket.send(chunk);
        }
      }
    }));
  } catch {}
}

function base64ToArray(b64Str) {
  if (!b64Str) return { error: null };
  try {
    b64Str = b64Str.replace(/-/g, '+').replace(/_/g, '/');
    return { earlyData: Uint8Array.from(atob(b64Str), c => c.charCodeAt(0)).buffer, error: null };
  } catch (error) {
    return { error };
  }
}

function closeSocketQuietly(socket) {
  try {
    if (socket.readyState === 1 || socket.readyState === 2) socket.close();
  } catch {}
}

function isValidFormat(uuid) {
  return /^[0-9a-f]{8}-[0-9a-f]{4}-4[0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i.test(uuid);
}

const hexTable = Array.from({ length: 256 }, (_, i) => (i + 256).toString(16).slice(1));

function formatIdentifier(arr, offset = 0) {
  const id = (
    hexTable[arr[offset]] + hexTable[arr[offset + 1]] + hexTable[arr[offset + 2]] + hexTable[arr[offset + 3]] + '-' +
    hexTable[arr[offset + 4]] + hexTable[arr[offset + 5]] + '-' +
    hexTable[arr[offset + 6]] + hexTable[arr[offset + 7]] + '-' +
    hexTable[arr[offset + 8]] + hexTable[arr[offset + 9]] + '-' +
    hexTable[arr[offset + 10]] + hexTable[arr[offset + 11]] + hexTable[arr[offset + 12]] + hexTable[arr[offset + 13]] +
    hexTable[arr[offset + 14]] + hexTable[arr[offset + 15]]
  ).toLowerCase();

  if (!isValidFormat(id)) throw new TypeError(E_INVALID_ID_STR);
  return id;
}
